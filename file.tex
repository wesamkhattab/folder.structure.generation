\newcommand\myfolder[2][fblue]{%
\begin{tikzpicture}[overlay]
\begin{scope}
\filldraw[draw=folderborder,top color=#1!50,bottom color=#1]
  (-1.05*\Size,0.2\Size+5pt) rectangle ++(.75*\Size,-0.2\Size-5pt);  
\filldraw[draw=folderborder,top color=#1!50,bottom color=#1]
   (-1.15*\Size,-\Size) rectangle (1.15*\Size,\Size);
\end{scope}  
\end{tikzpicture}%
\makebox{\raisebox{-3pt}{{\ttfamily\ /#2}}}%
}

\newcommand\myfile[2][fwhite]{%
\begin{tikzpicture}[overlay]
\begin{scope}
\filldraw[draw=folderborder,top color=#1!50,bottom color=#1]
   (-0.75*\FileSize,-\FileSize) rectangle (0.75*\FileSize,\FileSize);
\draw[white, fill=white] ($(0.75*\FileSize,\FileSize)+(-3pt,0)$) --++ (3pt,0) --++ (0,-3pt) -- cycle;
\fill[black] ($(0.75*\FileSize,\FileSize)+(0,-3pt)$) --++ (0,-0.5pt) --++ (-3.5pt,0) --++ (0,3.5pt) --++ (0.5pt,0) -- cycle;
\end{scope}  
\end{tikzpicture}%
\makebox{\raisebox{-3pt}{{\ttfamily\ /#2}}}%
}
